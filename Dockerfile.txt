# docker-compose.yml builds from this file directly (`dockerfile: Dockerfile.txt`),
# so no rename is needed on the VM. The `.txt` keeps it an inert, editable copy on
# Windows, and because it's the git-tracked file Compose uses, `git pull` keeps it
# current. Docker ignores the extension.

FROM node:22-alpine AS base

# Stage 1: Install dependencies (against the committed package-lock.json)
FROM base AS deps
WORKDIR /app
COPY package.json package-lock.json ./
RUN npm ci

# Stage 2: Build the application (produces .next/standalone)
FROM base AS builder
WORKDIR /app
COPY --from=deps /app/node_modules ./node_modules
COPY . .
RUN npm run build

# Stage 3: Minimal production server. 
FROM base AS runner
WORKDIR /app
ENV NODE_ENV=production
ENV PORT=3000 
ENV HOSTNAME=0.0.0.0

# --from=builder - Pull the following from the "builder" build stage
# /app/public - the src path in the build stage. 
# ./public - the destination path insid your final production container
COPY --from=builder /app/public ./public

COPY --from=builder /app/.next/standalone ./
COPY --from=builder /app/.next/static ./.next/static

EXPOSE 3000
CMD ["node", "server.js"]
